# Frontend URL (برای لینک‌های ورود/ثبت‌نام/محصولات)
# Development: http://localhost:3001
# Production: https://app.highwinrate.com (یا URL اصلی frontend)
NEXT_PUBLIC_FRONTEND_URL=http://localhost:3001

# Payload CMS Configuration (برای Blog)
# Secret key for Payload (generate a random string)
PAYLOAD_SECRET=your-secret-key-here

# Payload Server URL
# Development: http://localhost:3003
# Production: https://highwinrate.com
PAYLOAD_PUBLIC_SERVER_URL=http://localhost:3003

# Database Configuration (Supabase PostgreSQL)
# Option 1: Use connection string (recommended)
# POSTGRES_URL=postgresql://user:password@host:port/database

# Option 2: Use individual Supabase variables (preferred)
# Host can be extracted from SUPABASE_URL automatically, or set manually:
POSTGRES_HOST=db.xxxxxxxxxxxxx.supabase.co
POSTGRES_PORT=5432  # Optional, defaults to 5432
POSTGRES_USER=postgres  # Optional, defaults to 'postgres'
POSTGRES_PASSWORD=your-database-password  # REQUIRED!
POSTGRES_DATABASE=postgres  # Optional, defaults to 'postgres'

# Supabase URL (Public)
NEXT_PUBLIC_SUPABASE_URL=https://xxxxxxxxxxxxx.supabase.co

# Supabase Anon Key (Public)
NEXT_PUBLIC_SUPABASE_ANON_KEY=your-anon-key

# Alternative Supabase variables (also supported)
# SUPABASE_DB_URL=postgresql://user:password@host:port/database
